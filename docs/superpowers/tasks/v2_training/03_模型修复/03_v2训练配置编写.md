### Step 1: 基于 train_config_v2.yaml 创建最终配置
```yaml
# configs/train_v2.yaml
training:
  epochs: 50
  batch_size: 64
  lr: 0.001
  optimizer: AdamW
  weight_decay: 0.05
  lr_schedule: cosine
  warmup_epochs: 5
  grad_clip: 1.0
  amp: true              # 开启混合精度
  gradient_accumulation: 1
  early_stopping_patience: 15
  label_smoothing: 0.1

model:
  variant: "34"
  n_mels: 80
  embedding_dim: 192
  scale: 4
  se_reduction: 8
  pool_attention_dim: 128
  dropout: 0.3           # 实际使用

loss:
  type: aam_softmax
  margin: <T3.2 最优值>  # 从参数搜索结果填入
  scale: <T3.2 最优值>
  label_smoothing: 0.1

data:
  features_dir: data/voxceleb/features
  fixed_length: 200
  num_workers: 4
  pin_memory: true
  val_split_ratio: 0.1
  val_random_seed: 42
  augment: true            # 开启数据增强
  augment_config:
    spec_augment:
      time_mask: 10
      freq_mask: 20
      num_masks: 2
    random_gain:
      gain_range: 5.0

checkpoint:
  save_dir: checkpoints_v2
  save_best: true
  save_interval: 5
  resume: null             # 全新训练，不 resume

logging:
  tensorboard_dir: runs_v2
  log_interval: 50
  gpu_monitor_interval: 30
```
